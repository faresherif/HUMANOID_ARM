HIGH : LOW; while (OLD_ANGLE_1 != ANGLE) { digitalWrite(MOTOR_1_FORWARD, DIRECTION); digitalWrite(MOTOR_1_BACKWARD,!DIRECTION); OLD_ANGLE_1 += (DIRECTION == HIGH) ? 1 : -1; delay(10); } digitalWrite(MOTOR_1_FORWARD, LOW); digitalWrite(MOTOR_1_BACKWARD, LOW); } void READ_ENCODER_1(){ int B = digitalRead(ENCODER_B_1); if(B > 0){ POS_1++; } else{ POS_1--; } } void setup() { Serial.begin(9600); pinMode(ENCODER_A_1,INPUT); pinMode(ENCODER_B_1,INPUT); pinMode(MOTOR_1_FORWARD, OUTPUT); pinMode(MOTOR_1_BACKWARD, OUTPUT); attachInterrupt(digitalPinToInterrupt(ENCODER_A_1),READ_ENCODER_1,CHANGE); LCD.init(); //LCD I2C DATA PIN 20 , CLK PIN 21 LCD.backlight(); LCD.clear(); } void loop() { int POT_1_VALUE=analogRead(POT_1); int ANGLE_MOTOR_1=map(POT_1_VALUE,0,1023,MIN_ANGLE,MAX_ANGLE); MOVE_MOTOR_ANGLE_1(ANGLE_MOTOR_1); LCD.setCursor(0, 0); LCD.print("1st "); LCD.print(ANGLE_MOTOR_1); LCD.print((char)223); // Degree symbol LCD.setCursor(0, 1); LCD.print("pot"); LCD.print(POT_1_VALUE); delay(100); // Adjust delay as needed for display refresh rate }
